CREATE TRIGGER `ENCRYPT_INSERT` BEFORE INSERT ON `auth`
 FOR EACH ROW SET new.PASSWORD = AES_ENCRYPT(new.PASSWORD, 'nish')

CREATE TRIGGER `TRANSACT` AFTER INSERT ON `registration`
 FOR EACH ROW INSERT INTO transactions(R_ID, AMOUNT) VALUES(NEW.R_ID, (SELECT PRICE FROM events WHERE events.E_ID=NEW.E_ID))
